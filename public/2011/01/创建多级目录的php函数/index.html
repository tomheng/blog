//创建多级目录的PHP函数
function mmkdir(){
  $nums=func_num_args();
  if($nums>=1){
   $arglist=func_get_args();
   $rpath='.';
   foreach($arglist as $path){
    $rpath.='/'.$path;
     mkdir($rpath);
   }
 
  }

}
mmkdir('a','b');
