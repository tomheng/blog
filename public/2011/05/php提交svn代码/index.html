function executa($cmd, $pathInicial=null) {
    $resultado = array();
    $handle = popen("$cmd 2>&1", 'r');
    while ($read = fread($handle, 20096)) {
        $resultado[] = $read;
    }
    pclose($handle);
    flush();
    return $resultado;
}
